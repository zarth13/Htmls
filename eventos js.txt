EVENTOS JAVA SCRIPT 
Mouse
onMouseDown = clique do mouse pra baixo
onMouseUp   = clique do mouse pra cima
onMouseOver = quando mouse entrar na area da tag
onMouseOut  = quando mouse sai da area da tag
onMoseMove  = quando mouse movimentar em cima da tag
onClick = click '-'
oncontectmenu = click com botao direito (para bloquear a funçao do navegador : oncontectmenu="return funçao()" e na funçao coloca no final " return false;"
ondblclick = click duplo '-'

Teclado
onkeyup    = mesma coisa do mouse
onkeydown  = mesma coisa do mouse
onkeyprees = manter precionada a tecla
 usar (event) para capturar o evento ou seja a tecla clicada
  event.keyCode

Pagina
onload   = quado a pagina é carregada
onunload = quando a pagina é descarregada

Formulario
onchange = para select , quando a opçao for alterada  , temque mandar ele ex onchange="trocou(this)"
onfocus  = para campo de texto , quando ele estiver focado (clicado) ,
onblur   = contrario do focus
onsubmit = coloca no formulario , ele ativa quando o formulario for enviado
